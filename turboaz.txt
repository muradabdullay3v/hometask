CREATE DATABASE TURBOAZ

USE TURBOAZ



CREATE TABLE MARKA(
ID INT PRIMARY KEY IDENTITY,
NAME NVARCHAR(50) NOT NULL
)

CREATE TABLE MODEL(
ID INT PRIMARY KEY IDENTITY,
NAME NVARCHAR(50) NOT NULL,
MARKAID INT REFERENCES MARKA(ID)
)

CREATE TABLE BAN(
ID INT PRIMARY KEY IDENTITY,
NAME NVARCHAR(50) NOT NULL
)

CREATE TABLE RENG(
ID INT PRIMARY KEY IDENTITY,
NAME NVARCHAR(50) NOT NULL
)

CREATE TABLE YANACAQ(
ID INT PRIMARY KEY IDENTITY,
NAME NVARCHAR(50) NOT NULL
)

CREATE TABLE OTURUCU(
ID INT PRIMARY KEY IDENTITY,
NAME NVARCHAR(50) NOT NULL
)

CREATE TABLE TRANSMISSION(
ID INT PRIMARY KEY IDENTITY,
NAME NVARCHAR(50) NOT NULL
)

CREATE TABLE IMAGES(
ID INT PRIMARY KEY IDENTITY,
NAME NVARCHAR(50) NOT NULL
)

CREATE TABLE INFO(
ID INT PRIMARY KEY IDENTITY,
INFORMATION NVARCHAR(255) NOT NULL
)

CREATE TABLE KOMPLEKTASIYA(
ID INT PRIMARY KEY IDENTITY,
DISK BIT DEFAULT 0,
KOJA BIT DEFAULT 0,
ABSSYSTEM BIT DEFAULT 0,
RADAR BIT DEFAULT 0,
LYUK BIT DEFAULT 0,
CAM BIT DEFAULT 0
)


CREATE TABLE ELAN(
ID INT PRIMARY KEY IDENTITY,
MARKAID INT REFERENCES MARKA(ID),
BANID INT REFERENCES BAN(ID),
YURUSH FLOAT,
RENGID INT REFERENCES RENG(ID),
PRICE MONEY CHECK(PRICE > 0) NOT NULL,
YANACAQID INT REFERENCES YANACAQ(ID),
OTURUCUID INT REFERENCES OTURUCU(ID),
TRANSMISSIONID INT REFERENCES TRANSMISSION(ID),
IL INT CHECK(IL > 1952) NOT NULL,
MOTOR FLOAT CHECK (MOTOR >= 0) NOT NULL,
HP INT CHECK(HP > 0) NOT NULL,
KOPLEKTID INT REFERENCES KOMPLEKTASIYA(ID),
INFOID INT REFERENCES INFO(ID),
IMAGESID INT REFERENCES IMAGES(ID)
)
